 #!/bin/bash
if [ -z "$1" ]
then
echo "Error: no argument"
exit 1
fi
if [ -z "$2" ]
then
echo "Error: no argument"
exit 1
fi
if [ -z "$3" ]
then
echo "Error: no argument"
exit 1
fi
echo “Parameters exist”
#here we find in  $1 catalog files with $2 permission and change mod to $3
find $1 -perm $2 -exec chmod $3 '{}' \;
